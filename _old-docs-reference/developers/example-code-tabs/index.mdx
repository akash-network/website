# Multi-Language Code Examples

This page demonstrates the new CodeTabs component with multi-language support.

## Create a Deployment

Learn how to create a deployment on Akash Network using different programming languages.

<CodeTabs
  title="Create Deployment Example"
  examples={[
    {
      language: "python",
      code: `from akash import AkashClient

# Initialize the client
client = AkashClient(
    rpc_url="https://rpc.akash.network:443",
    chain_id="akashnet-2"
)

# Create deployment from SDL file
deployment = client.tx.create_deployment(
    sdl_path="deploy.yaml",
    from_key="mykey",
    deposit="5000000uakt"
)

print(f"Deployment created! DSEQ: {deployment.dseq}")
print(f"Status: {deployment.state}")`
    },
    {
      language: "javascript",
      code: `import { AkashClient } from '@akashnetwork/akashjs';

// Initialize the client
const client = new AkashClient({
  rpcUrl: 'https://rpc.akash.network:443',
  chainId: 'akashnet-2'
});

// Create deployment from SDL file
const deployment = await client.tx.createDeployment({
  sdlPath: 'deploy.yaml',
  fromKey: 'mykey',
  deposit: '5000000uakt'
});

console.log(\`Deployment created! DSEQ: \${deployment.dseq}\`);
console.log(\`Status: \${deployment.state}\`);`
    },
    {
      language: "typescript",
      code: `import { AkashClient, Deployment } from '@akashnetwork/akashjs';

// Initialize the client
const client = new AkashClient({
  rpcUrl: 'https://rpc.akash.network:443',
  chainId: 'akashnet-2'
});

// Create deployment from SDL file
const deployment: Deployment = await client.tx.createDeployment({
  sdlPath: 'deploy.yaml',
  fromKey: 'mykey',
  deposit: '5000000uakt'
});

console.log(\`Deployment created! DSEQ: \${deployment.dseq}\`);
console.log(\`Status: \${deployment.state}\`);`
    },
    {
      language: "go",
      code: `package main

import (
    "fmt"
    "log"
    
    akash "github.com/akash-network/akash-api/go/node/deployment/v1beta3"
)

func main() {
    // Initialize the client
    client := akash.NewClient(
        "https://rpc.akash.network:443",
        "akashnet-2",
    )
    
    // Create deployment from SDL file
    deployment, err := client.CreateDeployment(
        "deploy.yaml",
        "mykey",
        "5000000uakt",
    )
    if err != nil {
        log.Fatal(err)
    }
    
    fmt.Printf("Deployment created! DSEQ: %d\\n", deployment.Dseq)
    fmt.Printf("Status: %s\\n", deployment.State)
}`
    },
    {
      language: "bash",
      code: `# Set environment variables
export AKASH_KEY_NAME="mykey"
export AKASH_CHAIN_ID="akashnet-2"
export AKASH_NODE="https://rpc.akash.network:443"

# Create deployment
provider-services tx deployment create deploy.yaml \\
  --from $AKASH_KEY_NAME \\
  --chain-id $AKASH_CHAIN_ID \\
  --node $AKASH_NODE \\
  --deposit 5000000uakt

# Output:
# Deployment created! DSEQ: 1234567
# Status: active`
    }
  ]}
  client:load
/>

## Query Deployments

Query your existing deployments.

<CodeTabs
  title="Query Deployments"
  examples={[
    {
      language: "python",
      code: `# Query deployments for an owner
deployments = client.query.deployments(
    owner="akash1..."
)

for deployment in deployments:
    print(f"DSEQ: {deployment.dseq}")
    print(f"State: {deployment.state}")
    print(f"Created: {deployment.created_at}")
    print("---")`
    },
    {
      language: "javascript",
      code: `// Query deployments for an owner
const deployments = await client.query.deployments({
  owner: 'akash1...'
});

deployments.forEach(deployment => {
  console.log(\`DSEQ: \${deployment.dseq}\`);
  console.log(\`State: \${deployment.state}\`);
  console.log(\`Created: \${deployment.createdAt}\`);
  console.log('---');
});`
    },
    {
      language: "go",
      code: `// Query deployments for an owner
deployments, err := client.QueryDeployments(
    "akash1...",
)
if err != nil {
    log.Fatal(err)
}

for _, deployment := range deployments {
    fmt.Printf("DSEQ: %d\\n", deployment.Dseq)
    fmt.Printf("State: %s\\n", deployment.State)
    fmt.Printf("Created: %d\\n", deployment.CreatedAt)
    fmt.Println("---")
}`
    },
    {
      language: "bash",
      code: `# Query deployments
provider-services query deployment list \\
  --owner akash1... \\
  --node $AKASH_NODE

# Output shows list of deployments with DSEQ and status`
    }
  ]}
  client:load
