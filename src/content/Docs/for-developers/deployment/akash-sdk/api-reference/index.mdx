## Error Handling

<CodeTabs
  client:load
  examples={[
    {
      language: "go",
      code: `import "github.com/akash-network/akash-go/errors"

deployment, err := client.Deployments.Create(...)
if err != nil {
    if errors.IsInsufficientFunds(err) {
        log.Fatal("Not enough funds!")
    }
    log.Fatal(err)
}`,
    },
    {
      language: "javascript",
      label: "JavaScript/TypeScript",
      code: `import { AkashError, InsufficientFundsError } from '@akashnetwork/chain-sdk';

try {
  const deployment = await client.deployments.create(...);
} catch (error) {
  if (error instanceof InsufficientFundsError) {
    console.error('Not enough funds!');
  } else if (error instanceof AkashError) {
    console.error(\`Akash error: \${error.message}\`);
  }
}`,
    }]}
/>
